Напишите функцию, которая принимает массив чисел и возвращает массив уникальных дубликатов, которые были найдены


function findDuplicates(arr) { const duplicates = []; for (let i = 0; i < arr.length; i++) { for (let j = i + 1; j < arr.length; j++) { if (arr[i] === arr[j] && !duplicates.includes(arr[i])) { duplicates.push(arr[i]); } } } return duplicates; } const numbers = [1, 2, 7, 3, 4, , 7, 2, 5, 1, 2, 7]; console.log(findDuplicates(numbers)); // [1, 2]
